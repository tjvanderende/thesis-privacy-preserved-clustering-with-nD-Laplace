\nomenclature[P]{\(x \in X\)}{$X$ is the collection of original data-points belonging to a user.}
\nomenclature[P]{$z \in Z$}{$Z$ will be used to denote the output of a privacy mechanism}
\nomenclature[P]{$d(,,)$}{This function is used to calculate the Euclidean distance between two pairs of data-points}
\nomenclature[P]{$d_3(,,)$}{The Euclidean distance in 3-dimensional space.}
\nomenclature[P]{$\epsilon$}{This is the privacy budget of a differential privacy mechanism. A lower privacy budget, means there is more noise added}
\nomenclature[P]{$\theta$}{The theta is the angle of a data-point in the spherical coordinate system.}
\nomenclature[P]{$\psi$}{In 3-dimensional space, this is the other angle (besides $\theta$) in a spherical coordinate system.}
\nomenclature[P]{$r$}{This is the radial distance between an original point $x$ and the perturbed point $z$ .}
\nomenclature[P]{$G$}{This is a 2-dimensional grid.}
\nomenclature[P]{$G_c$}{This is a 3-dimensional cuboid grid.}
\nomenclature[P]{$G_n$}{This is a n-dimensional n-cube grid.}
\nomenclature[P]{$K(x_0)(x)$}{The $K$ is the probability of another point $x_0$ being close to $x$.}
\nomenclature[P]{$d_x$}{A generic distance function for calculating distance in n-dimensional space.}
\nomenclature[P]{$d_{euc}$}{The distance function for calculating the Euclidean distance in n-dimensional space.}
